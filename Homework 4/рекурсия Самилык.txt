	addi s0 s0 1 	# счетчик от 1 до ...
	addi s1 s1 1 	# текущее значение факториала для счетчика
	jal fact	# запускаем подпрограмму fact
	li a7 1        	# Системный вызов №1 - вывести целое число
	ecall
	li a7 10       	# Системный вызов №10 — остановка программы 
	ecall	
	
		
fact: 	addi sp sp -4	# запасаем 3 ячейки в стеке
	sw ra (sp)	# сохраняем ra
	
	addi s0 s0 1  	# увеличиваем счетчик на 1
	mv s2 s1   	# сохраняем в s2 текущее значение факториала 
	mul s1 s1 s0  	# считаем новое значение факториала
	div s3 s1 s0 	# получаем значение частного от делеия текущего факториала на текущее значение счетчика
	beq s3 s2 fact	# проверяем, поместился ли факториал в 32-х разрядном машинном слове
	j done
done:
	mv a0 s0	# в случае не похода проверки момещаем в регистр a0 значение счетчика
	addi a0 a0 -1	# уменьшаем значение в a0 на 1
	lw ra (sp)	# восстанавливаем ra
	addi sp sp 12	# восстанавливаем вершину стека
	ret
